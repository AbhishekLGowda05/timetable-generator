## PHASE 7: CP-SAT SOLVER IMPLEMENTATION

### File: `server/app/solver/model.py`

**Purpose**: Core constraint programming model.

**Critical Design Principles**:
1. **Global Solve**: All classes solved together in one model
2. **Variable Naming**: Use clear prefixes (X_, T_, R_, B_)
3. **Objective Function**: Weighted sum of soft constraint penalties
4. **Time Limit**: Configurable, default 30 seconds

**Must implement**:

```python
from ortools.sat.python import cp_model
from typing import Dict, List, Tuple

class TimetableSolver:
    def __init__(self, solver_input: dict):
        self.model = cp_model.CpModel()
        self.solver_input = solver_input
        self.variables = {}
        self.constraints_map = {}
        
    def build_model(self):
        """
        1. Create decision variables
        2. Add hard constraints
        3. Add soft constraints (with penalties)
        4. Set objective
        """
        
    def solve(self, time_limit: int = 30) -> dict:
        """
        Returns solver_output schema-compliant result
        """
        
    def extract_timetable(self, solver) -> dict:
        """
        Convert CP-SAT solution to timetable JSON
        """
```

**Variable Structure**:
```python
# Binary variables
X[section_id][day][period][subject_id] = 0 or 1
T[section_id][day][period][teacher_id] = 0 or 1
R[section_id][day][period][resource_id] = 0 or 1

# Block start indicators
B[section_id][subject_id][day][period] = 0 or 1

# Soft constraint slack variables
slack_teacher_balance[teacher_id] = integer >= 0
slack_gaps[teacher_id] = integer >= 0
```

**Objective Function**:
```python
minimize:
    w1 * sum(slack_teacher_balance) +
    w2 * sum(slack_gaps) +
    w3 * sum(core_after_lunch_violations) +
    w4 * sum(pe_early_violations)
```

- Be independently testable
- Handle toggle state from `solver_input.constraints`

**Must implement these functions**:

```python
def add_teacher_single_assignment_constraint(model, variables, teachers, period_grid):
    """
    For each teacher T and (day, period):
        sum over all sections: T[section][day][period][teacher_id] <= 1
    """

def add_max_consecutive_constraint(model, variables, teachers, period_grid):
    """
    For each teacher, for any window of (max_consecutive + 1) periods:
        sum of assignments <= max_consecutive
    """

def add_language_sync_constraint(model, variables, classes, period_grid):
    """
    If language_sync_enabled:
        For each class with language_block_enabled:
            For the designated language period:
                All language_teachers must be assigned
    """

def add_block_period_constraint(model, variables, subjects, period_grid):
    """
    For subjects requiring block periods:
        Use B[section][subject][day][start_period] indicator
        Force X[section][day][period] = 1 for next k periods
        Ensure no crossing of lunch/recess
    """

def add_resource_capacity_constraint(model, variables, resources, period_grid):
    """
    For each resource and (day, period):
        sum over sections using resource <= capacity
    """

def add_subject_frequency_constraint(model, variables, classes, subjects):
    """
    For each (section, subject):
        min_per_week <= sum of X[section][day][period][subject] <= max_per_week
    """

def add_teacher_load_bounds_constraint(model, variables, teachers, period_grid):
    """
    For each teacher:
        Daily: min_periods_day <= assignments <= max_periods_day
        Weekly: min_periods_week <= assignments <= max_periods_week
    """

def add_substitution_reserve_constraint(model, variables, teachers, period_grid, reserve_count):
    """
    For each period:
        count of unassigned teachers >= reserve_count
    """

def add_class_teacher_period_1_constraint(model, variables, classes, period_grid):
    """
    If enabled:
        For each class with class_teacher_id:
            T[section]["Mon"][1][class_teacher_id] = 1
            (and same for other days)
    """

# Soft constraints (return slack variables)

def add_core_morning_preference(model, variables, subjects, period_grid) -> list:
    """
    For core subjects scheduled after lunch:
        Create slack variable
    Returns: list of slack variables
    """

def add_teacher_balance_preference(model, variables, teachers, period_grid) -> list:
    """
    Minimize variance of teacher daily loads
    Returns: list of slack variables
    """

def add_minimize_gaps_preference(model, variables, teachers, period_grid) -> list:
    """
    Penalize gaps in teacher schedules
    Returns: list of slack variables
    """
```

## PHASE 10: API ENDPOINTS IMPLEMENTATION

### File: `server/app/api/solve.py`

```python
from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
import uuid

router = APIRouter()

class SolveRequest(BaseModel):
    school: dict
    classes: list[dict]
    teachers: list[dict]
    subjects: list[dict]
    resources: list[dict]
    constraints: dict
    time_limit_seconds: int = 30

@router.post("/solve")
async def solve_timetable(request: SolveRequest):
    """
    1. Validate solver_input schema
    2. Create job in JobQueue
    3. Start worker thread
    4. Return job_id immediately
    """
    # Validation
    solver_input = request.dict()
    is_valid, errors = validate_solver_input(solver_input)
    if not is_valid:
        raise HTTPException(400, {"errors": errors})
    
    # Pre-flight feasibility check
    is_feasible, warnings = validate_feasibility(solver_input)
    if not is_feasible:
        return {
            "status": "rejected",
            "warnings": warnings,
            "suggestion": "Fix data or relax constraints"
        }
    
    # Create job
    job_id = str(uuid.uuid4())
    job_queue.create_job(job_id, solver_input)
    
    # Start worker
    worker_thread = Thread(target=execute_solver_job, args=(job_id, solver_input))
    worker_thread.start()
    
    return {"job_id": job_id, "status": "queued"}
```

```python
async def validate_swap(request: SwapRequest):
    """
    1. Load existing timetable from job result
    2. Apply proposed swap
    3. Check hard constraints only (no re-solve)
    4. Return valid/invalid + reasons
    """
    job = job_queue.get_job(request.job_id)
    if not job or not job.result:
        raise HTTPException(404, "Timetable not found")
    
    timetable = job.result["timetable"]
    
    # Apply swap
    temp_timetable = apply_swap(timetable, request)
    
    # Validate
    violations = check_hard_constraints(temp_timetable, job.solver_input)
    
    if violations:
        return {
            "valid": False,
            "violations": violations
        }
    
    return {"valid": True}
```

```python
from fastapi import UploadFile, File

@router.post("/upload")
async def upload_demo_data(file: UploadFile = File(...), data_type: str = "teachers"):
    """
    1. Save to raw_uploads/
    2. Parse CSV/XLSX
    3. Validate against schema
    4. Save to parsed/
    5. Return parsed data + manifest_id
    """
    # Save raw file
    file_id = str(uuid.uuid4())
    raw_path = RAW_UPLOADS_DIR / f"{file_id}_{file.filename}"
    
    with open(raw_path, "wb") as f:
        f.write(await file.read())
    
    # Parse
    if file.filename.endswith(".csv"):
        parsed = parse_csv(raw_path, data_type)
    elif file.filename.endswith(".xlsx"):
        parsed = parse_excel(raw_path, data_type)
    else:
        raise HTTPException(400, "Unsupported format")
    
    # Validate
    validator = get_validator(data_type)
    is_valid, errors = validator(parsed)
    if not is_valid:
        return {
            "status": "error",
            "errors": errors
        }
    
    # Save parsed
    parsed_path = PARSED_DIR / f"{file_id}_{data_type}.json"
    with open(parsed_path, "w") as f:
        json.dump(parsed, f, indent=2)
    
    return {
        "status": "success",
        "manifest_id": file_id,
        "parsed_count": len(parsed)
    }
```
